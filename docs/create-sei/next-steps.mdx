After scaffolding your Sei dApp with `@sei-js/create-sei`, this guide helps you understand your project structure and continue development with template-specific patterns and best practices.

## Project Setup

After creating your project, complete the initial setup:

<Steps>
  <Step title="Navigate to Project">
    ```bash
    cd my-sei-app
    ```
  </Step>
  
  <Step title="Install Dependencies">
    ```bash
    npm install
    # or
    yarn install
    # or
    pnpm install
    ```
  </Step>
  
  <Step title="Configure Environment">
    Copy the example environment file and add your configuration:
    ```bash
    cp .env.example .env.local
    ```
    
    Edit `.env.local` with your settings:
    ```env
    NEXT_PUBLIC_WALLETCONNECT_PROJECT_ID=your_project_id
    NEXT_PUBLIC_SEI_RPC_URL=https://evm-rpc.sei-apis.com
    NEXT_PUBLIC_SEI_CHAIN_ID=1329
    ```
  </Step>
  
  <Step title="Start Development Server">
    ```bash
    npm run dev
    # or
    yarn dev
    # or
    pnpm dev
    ```
    
    Your app will be available at `http://localhost:3000`
  </Step>
</Steps>

<AccordionGroup>
  <Accordion title="Wallet Connection">
    The template includes a complete wallet connection setup with Wagmi:
    
    ```tsx
    // src/components/wallet/ConnectWallet.tsx
    import { useAccount, useConnect, useDisconnect } from 'wagmi'
    
    export function ConnectWallet() {
      const { address, isConnected } = useAccount()
      const { connect, connectors } = useConnect()
      const { disconnect } = useDisconnect()
      
      if (isConnected) {
        return (
          <div className="flex items-center gap-4">
            <span className="text-sm">
              {address?.slice(0, 6)}...{address?.slice(-4)}
            </span>
            <button onClick={() => disconnect()}>
              Disconnect
            </button>
          </div>
        )
      }
      
      return (
        <div className="space-y-2">
          {connectors.map((connector) => (
            <button
              key={connector.id}
              onClick={() => connect({ connector })}
              className="btn-primary"
            >
              Connect {connector.name}
            </button>
          ))}
        </div>
      )
    }
    ```
  </Accordion>
  
  <Accordion title="Contract Interactions">
    Use Wagmi hooks for type-safe contract interactions:
    
    ```tsx
    // src/hooks/useTokenBalance.ts
    import { useReadContract } from 'wagmi'
    import { erc20Abi } from 'viem'
    
    export function useTokenBalance(
      tokenAddress: `0x${string}`,
      userAddress: `0x${string}` | undefined
    ) {
      return useReadContract({
        address: tokenAddress,
        abi: erc20Abi,
        functionName: 'balanceOf',
        args: userAddress ? [userAddress] : undefined,
        query: {
          enabled: !!userAddress,
        },
      })
    }
    ```
  </Accordion>

**Key Development Patterns:**

<AccordionGroup>
  <Accordion title="Wallet Management">
    Custom wallet connection with Ethers.js:
    
    ```tsx
    // src/hooks/useWallet.ts
    import { useState, useEffect } from 'react'
    import { BrowserProvider, JsonRpcSigner } from 'ethers'
    
    export function useWallet() {
      const [provider, setProvider] = useState<BrowserProvider | null>(null)
      const [signer, setSigner] = useState<JsonRpcSigner | null>(null)
      const [address, setAddress] = useState<string | null>(null)
      
      const connect = async () => {
        if (typeof window.ethereum !== 'undefined') {
          const provider = new BrowserProvider(window.ethereum)
          const signer = await provider.getSigner()
          const address = await signer.getAddress()
          
          setProvider(provider)
          setSigner(signer)
          setAddress(address)
        }
      }
      
      return { provider, signer, address, connect }
    }
    ```
  </Accordion>
  
  <Accordion title="Sei Precompiles">
    Direct access to Sei's native blockchain functions:
    
    ```tsx
    // src/hooks/useSeiPrecompiles.ts
    import { useWallet } from './useWallet'
    import { Contract } from 'ethers'
    import { BANK_PRECOMPILE_ADDRESS, BANK_ABI } from '../lib/precompiles'
    
    export function useBankPrecompile() {
      const { signer } = useWallet()
      
      const getBalance = async (denom: string) => {
        if (!signer) return null
        
        const contract = new Contract(
          BANK_PRECOMPILE_ADDRESS,
          BANK_ABI,
          signer
        )
        
        const address = await signer.getAddress()
        return await contract.balance(address, denom)
      }
      
      const sendTokens = async (to: string, amount: string, denom: string) => {
        if (!signer) return null
        
        const contract = new Contract(
          BANK_PRECOMPILE_ADDRESS,
          BANK_ABI,
          signer
        )
        
        return await contract.send(to, amount, denom)
      }
      
      return { getBalance, sendTokens }
    }
    ```
  </Accordion>
</AccordionGroup>

<AccordionGroup>
  <Accordion title="Lightweight Setup">
    Minimal configuration for rapid development:
    
    ```tsx
    // src/lib/ethers.ts
    import { BrowserProvider } from 'ethers'
    
    export const getSeiProvider = () => {
      if (typeof window.ethereum !== 'undefined') {
        return new BrowserProvider(window.ethereum)
      }
      throw new Error('No wallet found')
    }
    
    export const getSeiSigner = async () => {
      const provider = getSeiProvider()
      return await provider.getSigner()
    }
    ```
  </Accordion>
  
  <Accordion title="Direct Contract Access">
    Direct contract interactions without abstractions:
    
    ```tsx
    // src/hooks/useContract.ts
    import { useState, useEffect } from 'react'
    import { Contract } from 'ethers'
    import { getSeiSigner } from '../lib/ethers'
    
    export function useContract(address: string, abi: any[]) {
      const [contract, setContract] = useState<Contract | null>(null)
      
      useEffect(() => {
        const initContract = async () => {
          try {
            const signer = await getSeiSigner()
            const contractInstance = new Contract(address, abi, signer)
            setContract(contractInstance)
          } catch (error) {
            console.error('Failed to initialize contract:', error)
          }
        }
        
        initContract()
      }, [address, abi])
      
      return contract
    }
    ```
  </Accordion>
</AccordionGroup>

## Common Customizations

### Adding State Management

<AccordionGroup>
  <Accordion title="Zustand Store">
    Add global state management:
    
    ```bash
    npm install zustand
    ```
    
    ```tsx
    // src/store/useAppStore.ts
    import { create } from 'zustand'
    import { persist } from 'zustand/middleware'
    
    interface AppState {
      theme: 'light' | 'dark'
      setTheme: (theme: 'light' | 'dark') => void
      favoriteTokens: string[]
      addFavoriteToken: (address: string) => void
    }
    
    export const useAppStore = create<AppState>()(
      persist(
        (set) => ({
          theme: 'light',
          setTheme: (theme) => set({ theme }),
          favoriteTokens: [],
          addFavoriteToken: (address) =>
            set((state) => ({
              favoriteTokens: [...state.favoriteTokens, address]
            })),
        }),
        { name: 'sei-app-storage' }
      )
    )
    ```
  </Accordion>
  
  <Accordion title="Error Boundaries">
    Add comprehensive error handling:
    
    ```tsx
    // src/components/ErrorBoundary.tsx
    import React from 'react'
    
    interface ErrorBoundaryState {
      hasError: boolean
      error?: Error
    }
    
    export class ErrorBoundary extends React.Component<
      React.PropsWithChildren<{}>,
      ErrorBoundaryState
    > {
      constructor(props: React.PropsWithChildren<{}>) {
        super(props)
        this.state = { hasError: false }
      }
    
      static getDerivedStateFromError(error: Error): ErrorBoundaryState {
        return { hasError: true, error }
      }
    
      render() {
        if (this.state.hasError) {
          return (
            <div className="min-h-screen flex items-center justify-center">
              <div className="text-center">
                <h2 className="text-2xl font-bold text-red-600 mb-4">
                  Something went wrong
                </h2>
                <button
                  onClick={() => window.location.reload()}
                  className="btn-primary"
                >
                  Reload Page
                </button>
              </div>
            </div>
          )
        }
    
        return this.props.children
      }
    }
    ```
  </Accordion>
</AccordionGroup>

    Create reusable component variants:
    
    ```tsx
    // src/components/ui/Button.tsx
    import { cva, type VariantProps } from 'class-variance-authority'
    import { cn } from '@/lib/utils'
    
    const buttonVariants = cva(
      'inline-flex items-center justify-center rounded-md font-medium transition-colors',
      {
        variants: {
          variant: {
            default: 'bg-sei-500 text-white hover:bg-sei-600',
            outline: 'border border-sei-500 text-sei-500 hover:bg-sei-50',
            ghost: 'text-sei-500 hover:bg-sei-50',
          },
          size: {
            sm: 'h-8 px-3 text-sm',
            md: 'h-10 px-4',
            lg: 'h-12 px-6 text-lg',
          },
        },
        defaultVariants: {
          variant: 'default',
          size: 'md',
        },
      }
    )
    
    interface ButtonProps
      extends React.ButtonHTMLAttributes<HTMLButtonElement>,
        VariantProps<typeof buttonVariants> {}
    
    export function Button({ className, variant, size, ...props }: ButtonProps) {
      return (
        <button
          className={cn(buttonVariants({ variant, size, className }))}
          {...props}
        />
      )
    }
    ```
