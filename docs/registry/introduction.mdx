## Overview

The `@sei-js/registry` package provides TypeScript typed exports for the Sei registry repository and community asset lists. It includes comprehensive information about tokens, networks, IBC configurations, and gas settings for the Sei ecosystem.

## Quick Start

<CodeGroup>

```typescript Basic Usage
import { TOKEN_LIST, NETWORKS, IBC_INFO, GAS_INFO } from '@sei-js/registry';

// Find a specific token
const uAtom = TOKEN_LIST.find(asset => asset.denom === 'uatom');
console.log('ATOM token info:', uAtom);

// Get network information
const seiMainnet = NETWORKS.find(network => network.chain_id === 'pacific-1');
console.log('Sei mainnet info:', seiMainnet);

// Access IBC information
console.log('IBC configurations:', IBC_INFO);

// Get gas information
console.log('Gas prices:', GAS_INFO);
```

```typescript Token Lookup
import { TOKEN_LIST } from '@sei-js/registry';

// Find token by denom
function findTokenByDenom(denom: string) {
  return TOKEN_LIST.find(token => 
    token.denom === denom || 
    token.symbol.toLowerCase() === denom.toLowerCase()
  );
}

// Find tokens by chain
function findTokensByChain(chainId: string) {
  return TOKEN_LIST.filter(token => 
    token.traces?.some(trace => trace.chain?.chain_id === chainId)
  );
}

// Usage
const seiToken = findTokenByDenom('usei');
const cosmosTokens = findTokensByChain('cosmoshub-4');
```

</CodeGroup>

## Token Registry

### TOKEN_LIST

The `TOKEN_LIST` export contains comprehensive information about all supported tokens in the Sei ecosystem.

<CodeGroup>

```typescript Token Structure
interface Token {
  denom: string;           // Base denomination
  symbol: string;          // Display symbol
  name: string;           // Full name
  decimals: number;       // Decimal places
  logo_URIs?: {           // Logo images
    png?: string;
    svg?: string;
  };
  coingecko_id?: string;  // CoinGecko identifier
  traces?: TraceInfo[];   // IBC trace information
  verified?: boolean;     // Verification status
  unstable?: boolean;     // Stability flag
}

// Example usage
const tokens = TOKEN_LIST.filter(token => token.verified === true);
console.log('Verified tokens:', tokens);
```

```typescript Common Tokens
import { TOKEN_LIST } from '@sei-js/registry';

// Sei native token
const SEI = TOKEN_LIST.find(t => t.symbol === 'SEI');

// Popular IBC tokens
const ATOM = TOKEN_LIST.find(t => t.symbol === 'ATOM');
const OSMO = TOKEN_LIST.find(t => t.symbol === 'OSMO');
const USDC = TOKEN_LIST.find(t => t.symbol === 'USDC');

// Factory tokens
const factoryTokens = TOKEN_LIST.filter(t => 
  t.denom.startsWith('factory/')
);

console.log('Factory tokens:', factoryTokens);
```

</CodeGroup>

## Network Registry

### NETWORKS

The `NETWORKS` export provides comprehensive network and chain information.

<CodeGroup>

```typescript Network Structure
interface Network {
  chain_id: string;        // Chain identifier
  chain_name: string;      // Human-readable name
  status: string;          // Network status
  network_type: string;    // mainnet/testnet/devnet
  pretty_name: string;     // Display name
  bech32_prefix: string;   // Address prefix
  daemon_name: string;     // Binary name
  node_home: string;       // Node home directory
  key_algos: string[];     // Supported algorithms
  slip44: number;          // SLIP-0044 coin type
  fees: FeeInfo;           // Fee information
  staking: StakingInfo;    // Staking parameters
  codebase: CodebaseInfo;  // Software versions
  apis: ApiEndpoints;      // RPC/REST endpoints
  explorers: Explorer[];   // Block explorers
}

// Example usage
const mainnet = NETWORKS.find(n => n.chain_id === 'pacific-1');
const testnet = NETWORKS.find(n => n.chain_id === 'atlantic-2');
```

```typescript Network Utilities
import { NETWORKS } from '@sei-js/registry';

// Get RPC endpoints
function getRpcEndpoints(chainId: string) {
  const network = NETWORKS.find(n => n.chain_id === chainId);
  return network?.apis?.rpc?.map(rpc => rpc.address) || [];
}

// Get REST endpoints
function getRestEndpoints(chainId: string) {
  const network = NETWORKS.find(n => n.chain_id === chainId);
  return network?.apis?.rest?.map(rest => rest.address) || [];
}

// Get network by type
function getNetworksByType(type: 'mainnet' | 'testnet' | 'devnet') {
  return NETWORKS.filter(n => n.network_type === type);
}

// Usage
const mainnetRpcs = getRpcEndpoints('pacific-1');
const testnetRests = getRestEndpoints('atlantic-2');
const mainnets = getNetworksByType('mainnet');
```

</CodeGroup>

## IBC Configuration

### IBC_INFO

The `IBC_INFO` export contains Inter-Blockchain Communication configurations.

<CodeGroup>

```typescript IBC Structure
interface IBCInfo {
  chain_1: {
    chain_name: string;
    client_id: string;
    connection_id: string;
  };
  chain_2: {
    chain_name: string;
    client_id: string;
    connection_id: string;
  };
  channels: Channel[];
}

interface Channel {
  chain_1: {
    channel_id: string;
    port_id: string;
  };
  chain_2: {
    channel_id: string;
    port_id: string;
  };
  ordering: string;
  version: string;
  tags?: {
    status: string;
    preferred: boolean;
    dex?: string;
  };
}

// Example usage
const seiCosmosIBC = IBC_INFO.find(ibc => 
  (ibc.chain_1.chain_name === 'sei' && ibc.chain_2.chain_name === 'cosmoshub') ||
  (ibc.chain_1.chain_name === 'cosmoshub' && ibc.chain_2.chain_name === 'sei')
);
```

```typescript IBC Utilities
import { IBC_INFO } from '@sei-js/registry';

// Find IBC connection between chains
function findIBCConnection(chain1: string, chain2: string) {
  return IBC_INFO.find(ibc => 
    (ibc.chain_1.chain_name === chain1 && ibc.chain_2.chain_name === chain2) ||
    (ibc.chain_1.chain_name === chain2 && ibc.chain_2.chain_name === chain1)
  );
}

// Get transfer channels
function getTransferChannels(chain1: string, chain2: string) {
  const connection = findIBCConnection(chain1, chain2);
  return connection?.channels.filter(ch => 
    ch.chain_1.port_id === 'transfer' && ch.chain_2.port_id === 'transfer'
  ) || [];
}

// Get preferred channel
function getPreferredChannel(chain1: string, chain2: string) {
  const channels = getTransferChannels(chain1, chain2);
  return channels.find(ch => ch.tags?.preferred === true) || channels[0];
}

// Usage
const seiOsmosisChannel = getPreferredChannel('sei', 'osmosis');
console.log('Preferred channel:', seiOsmosisChannel);
```

</CodeGroup>

## Gas Information

### GAS_INFO

The `GAS_INFO` export provides transaction fee and gas price information.

<CodeGroup>

```typescript Gas Structure
interface GasInfo {
  [chainId: string]: {
    low: number;      // Low gas price
    average: number;  // Average gas price  
    high: number;     // High gas price
    denom: string;    // Fee denomination
  };
}

// Example usage
import { GAS_INFO } from '@sei-js/registry';

const seiGasPrices = GAS_INFO['pacific-1'];
console.log('Sei gas prices:', seiGasPrices);

// Calculate transaction fee
function calculateFee(gasUsed: number, gasPrice: 'low' | 'average' | 'high' = 'average') {
  const chainGasInfo = GAS_INFO['pacific-1'];
  const pricePerGas = chainGasInfo[gasPrice];
  
  return {
    amount: Math.ceil(gasUsed * pricePerGas).toString(),
    denom: chainGasInfo.denom
  };
}

// Usage
const fee = calculateFee(200000, 'average');
console.log('Transaction fee:', fee);
```

```typescript Fee Utilities
import { GAS_INFO } from '@sei-js/registry';

// Get gas prices for chain
function getGasPrices(chainId: string) {
  return GAS_INFO[chainId] || null;
}

// Create fee object for CosmJS
function createStdFee(gasLimit: number, chainId: string, speed: 'low' | 'average' | 'high' = 'average') {
  const gasInfo = getGasPrices(chainId);
  if (!gasInfo) throw new Error(`Gas info not found for chain ${chainId}`);
  
  const gasPrice = gasInfo[speed];
  const amount = Math.ceil(gasLimit * gasPrice);
  
  return {
    amount: [{ denom: gasInfo.denom, amount: amount.toString() }],
    gas: gasLimit.toString()
  };
}

// Usage
const fee = createStdFee(200000, 'pacific-1', 'average');
```

## Advanced Usage

### Token Price Integration

<CodeGroup>

```typescript Price Lookup
import { TOKEN_LIST } from '@sei-js/registry';

// Get tokens with CoinGecko IDs
const tokensWithPrices = TOKEN_LIST.filter(token => token.coingecko_id);

// Fetch prices from CoinGecko
async function fetchTokenPrices(tokenIds: string[]) {
  const response = await fetch(
    `https://api.coingecko.com/api/v3/simple/price?ids=${tokenIds.join(',')}&vs_currencies=usd`
  );
  return response.json();
}

// Get current prices
async function getCurrentPrices() {
  const tokenIds = tokensWithPrices.map(t => t.coingecko_id!);
  const prices = await fetchTokenPrices(tokenIds);
  
  return tokensWithPrices.map(token => ({
    ...token,
    price_usd: prices[token.coingecko_id!]?.usd || null
  }));
}
```

```typescript Portfolio Calculation
// Calculate portfolio value
function calculatePortfolioValue(balances: Array<{denom: string, amount: string}>, prices: any) {
  return balances.reduce((total, balance) => {
    const token = TOKEN_LIST.find(t => t.denom === balance.denom);
    if (!token?.coingecko_id) return total;
    
    const price = prices[token.coingecko_id]?.usd || 0;
    const amount = parseFloat(balance.amount) / Math.pow(10, token.decimals);
    
    return total + (amount * price);
  }, 0);
}
```

## Validation and Type Safety

<CodeGroup>

```typescript Type Guards
import { TOKEN_LIST } from '@sei-js/registry';

// Type guard for verified tokens
function isVerifiedToken(token: any): token is VerifiedToken {
  return token && token.verified === true && token.symbol && token.denom;
}

// Validate token existence
function validateToken(denom: string): boolean {
  return TOKEN_LIST.some(token => token.denom === denom);
}

// Safe token lookup
function safeTokenLookup(denom: string) {
  const token = TOKEN_LIST.find(t => t.denom === denom);
  if (!token) {
    throw new Error(`Token not found: ${denom}`);
  }
  return token;
}

// Usage
try {
  const token = safeTokenLookup('usei');
  if (isVerifiedToken(token)) {
    console.log('Verified token:', token);
  }
} catch (error) {
  console.error('Token lookup failed:', error.message);
}
```

```typescript Registry Validation
// Validate network configuration
function validateNetwork(chainId: string) {
  const network = NETWORKS.find(n => n.chain_id === chainId);
  if (!network) {
    throw new Error(`Network not found: ${chainId}`);
  }
  
  if (!network.apis?.rpc?.length) {
    throw new Error(`No RPC endpoints for network: ${chainId}`);
  }
  
  return network;
}

// Validate IBC path
function validateIBCPath(sourceChain: string, destChain: string) {
  const connection = IBC_INFO.find(ibc => 
    (ibc.chain_1.chain_name === sourceChain && ibc.chain_2.chain_name === destChain) ||
    (ibc.chain_1.chain_name === destChain && ibc.chain_2.chain_name === sourceChain)
  );
  
  if (!connection) {
    throw new Error(`No IBC connection between ${sourceChain} and ${destChain}`);
  }
  
  return connection;
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Token Handling">
    - Always check token verification status
    - Use proper decimal conversion for amounts
    - Validate token existence before operations
    - Handle missing logo URIs gracefully
  </Accordion>

  <Accordion title="Network Configuration">
    - Use multiple RPC endpoints for redundancy
    - Implement fallback mechanisms for failed endpoints
    - Cache network information to reduce lookups
    - Validate chain IDs before transactions
  </Accordion>

  <Accordion title="IBC Operations">
    - Verify IBC connections before transfers
    - Use preferred channels when available
    - Handle channel updates and deprecations
    - Implement proper timeout handling
  </Accordion>

  <Accordion title="Gas Management">
    - Use appropriate gas prices for transaction priority
    - Implement dynamic gas estimation
    - Handle gas price fluctuations
    - Provide user options for fee preferences
  </Accordion>
</AccordionGroup>
