## Overview

The `@sei-js/sei-global-wallet` package provides [Dynamic Global Wallet](https://docs.dynamic.xyz/global-wallets/overview) support for Sei blockchain. It enables seamless wallet integration using the EIP-6963 standard, allowing users to connect and interact with Sei through a unified wallet interface.

## EIP-6963 Standard

### What is EIP-6963?

EIP-6963 is an Ethereum Improvement Proposal that standardizes wallet discovery for web applications. It allows wallets to announce their presence to web applications in a standardized way, enabling better wallet detection and user experience.

## Usage Patterns

### With React Applications

<CodeGroup>

```jsx React Hook Example
import "@sei-js/sei-global-wallet/eip6963";
import { useEffect, useState } from 'react';

function WalletConnection() {
  const [wallets, setWallets] = useState([]);
  const [selectedWallet, setSelectedWallet] = useState(null);

  useEffect(() => {
    // Listen for wallet announcements
    const handleWalletAnnouncement = (event) => {
      setWallets(prev => [...prev, event.detail]);
    };

    window.addEventListener('eip6963:announceProvider', handleWalletAnnouncement);

    // Request wallet announcements
    window.dispatchEvent(new Event('eip6963:requestProvider'));

    return () => {
      window.removeEventListener('eip6963:announceProvider', handleWalletAnnouncement);
    };
  }, []);

  const connectWallet = async (wallet) => {
    try {
      const accounts = await wallet.provider.request({
        method: 'eth_requestAccounts'
      });
      setSelectedWallet({ ...wallet, accounts });
    } catch (error) {
      console.error('Failed to connect wallet:', error);
    }
  };

  return (
    <div>
      <h3>Available Wallets</h3>
      {wallets.map((wallet) => (
        <button
          key={wallet.info.uuid}
          onClick={() => connectWallet(wallet)}
          className="wallet-option"
        >
          <img src={wallet.info.icon} alt={wallet.info.name} width="24" />
          {wallet.info.name}
        </button>
      ))}
      
      {selectedWallet && (
        <div>
          <h4>Connected to {selectedWallet.info.name}</h4>
          <p>Account: {selectedWallet.accounts[0]}</p>
        </div>
      )}
    </div>
  );
}
```

```jsx With Context Provider
import "@sei-js/sei-global-wallet/eip6963";
import { createContext, useContext, useEffect, useState } from 'react';

const WalletContext = createContext();

export function WalletProvider({ children }) {
  const [wallets, setWallets] = useState([]);
  const [activeWallet, setActiveWallet] = useState(null);

  useEffect(() => {
    const handleAnnouncement = (event) => {
      const wallet = event.detail;
      setWallets(prev => {
        const exists = prev.find(w => w.info.uuid === wallet.info.uuid);
        return exists ? prev : [...prev, wallet];
      });
    };

    window.addEventListener('eip6963:announceProvider', handleAnnouncement);
    window.dispatchEvent(new Event('eip6963:requestProvider'));

    return () => {
      window.removeEventListener('eip6963:announceProvider', handleAnnouncement);
    };
  }, []);

  const connectWallet = async (wallet) => {
    try {
      await wallet.provider.request({ method: 'eth_requestAccounts' });
      setActiveWallet(wallet);
    } catch (error) {
      console.error('Connection failed:', error);
    }
  };

  const disconnectWallet = () => {
    setActiveWallet(null);
  };

  return (
    <WalletContext.Provider value={{
      wallets,
      activeWallet,
      connectWallet,
      disconnectWallet
    }}>
      {children}
    </WalletContext.Provider>
  );
}

export const useWallet = () => useContext(WalletContext);
```

### With Vanilla JavaScript

<CodeGroup>

```javascript Wallet Discovery
import "@sei-js/sei-global-wallet/eip6963";

class WalletManager {
  constructor() {
    this.wallets = new Map();
    this.activeWallet = null;
    this.init();
  }

  init() {
    // Listen for wallet announcements
    window.addEventListener('eip6963:announceProvider', (event) => {
      const wallet = event.detail;
      this.wallets.set(wallet.info.uuid, wallet);
      this.onWalletDiscovered(wallet);
    });

    // Request wallet announcements
    window.dispatchEvent(new Event('eip6963:requestProvider'));
  }

  onWalletDiscovered(wallet) {
    console.log('Discovered wallet:', wallet.info.name);
    // Update UI with new wallet option
  }

  async connectWallet(uuid) {
    const wallet = this.wallets.get(uuid);
    if (!wallet) throw new Error('Wallet not found');

    try {
      const accounts = await wallet.provider.request({
        method: 'eth_requestAccounts'
      });
      
      this.activeWallet = { ...wallet, accounts };
      return this.activeWallet;
    } catch (error) {
      throw new Error(`Failed to connect: ${error.message}`);
    }
  }

  async sendTransaction(transaction) {
    if (!this.activeWallet) throw new Error('No wallet connected');

    return await this.activeWallet.provider.request({
      method: 'eth_sendTransaction',
      params: [transaction]
    });
  }
}

// Usage
const walletManager = new WalletManager();
```

import "@sei-js/sei-global-wallet/eip6963";

async function handleSeiTransaction() {
  // Wait for wallet discovery
  await new Promise(resolve => {
    const checkWallets = () => {
      if (window.seiGlobalWallet) {
        resolve();
      } else {
        setTimeout(checkWallets, 100);
      }
    };
    checkWallets();
  });

  try {
    // Connect to Sei Global Wallet
    const accounts = await window.seiGlobalWallet.request({
      method: 'eth_requestAccounts'

    // Send transaction
    const txHash = await window.seiGlobalWallet.request({

      params: [{
        from: accounts[0],
        to: '0x...',
        value: '0x1000000000000000', // 0.001 SEI
        gas: '0x5208',
        gasPrice: '0x3b9aca00'
      }]
    });

    console.log('Transaction sent:', txHash);
  } catch (error) {
    console.error('Transaction failed:', error);
  }
}

### Custom Wallet Provider

<CodeGroup>

```javascript Custom Provider Wrapper
import "@sei-js/sei-global-wallet/eip6963";

class SeiWalletProvider {
  constructor() {
    this.provider = null;
    this.accounts = [];
    this.chainId = null;
    this.init();
  }

  async init() {
    // Wait for Sei Global Wallet
    await this.waitForWallet();
    
    // Set up event listeners
    this.provider.on('accountsChanged', this.handleAccountsChanged.bind(this));
    this.provider.on('chainChanged', this.handleChainChanged.bind(this));
    this.provider.on('disconnect', this.handleDisconnect.bind(this));
  }

  async waitForWallet() {
    return new Promise((resolve) => {
      const checkWallet = () => {
        const wallets = this.getAvailableWallets();
        const seiWallet = wallets.find(w => w.info.name.includes('Sei'));
        
        if (seiWallet) {
          this.provider = seiWallet.provider;
          resolve();
        } else {
          setTimeout(checkWallet, 100);
        }
      };
      checkWallet();
    });
  }

  getAvailableWallets() {
    const wallets = [];
    window.addEventListener('eip6963:announceProvider', (event) => {
      wallets.push(event.detail);
    });
    window.dispatchEvent(new Event('eip6963:requestProvider'));
    return wallets;
  }

  async connect() {
    if (!this.provider) throw new Error('Wallet not initialized');
    
    this.accounts = await this.provider.request({
      method: 'eth_requestAccounts'
    });
    
    this.chainId = await this.provider.request({
      method: 'eth_chainId'
    });
    
    return { accounts: this.accounts, chainId: this.chainId };
  }

  async switchToSei() {
    try {
      await this.provider.request({
        method: 'wallet_switchEthereumChain',
        params: [{ chainId: '0x531' }], // Sei mainnet
      });
    } catch (error) {
      if (error.code === 4902) {
        // Chain not added, add it
        await this.addSeiNetwork();
      } else {
        throw error;
      }
    }
  }

  async addSeiNetwork() {
    await this.provider.request({
      method: 'wallet_addEthereumChain',
      params: [{
        chainId: '0x531',
        chainName: 'Sei Network',
        nativeCurrency: {
          name: 'SEI',
          symbol: 'SEI',
          decimals: 18
        },
        rpcUrls: ['https://evm-rpc.sei-apis.com'],
        blockExplorerUrls: ['https://seitrace.com']
      }]
    });
  }

  handleAccountsChanged(accounts) {
    this.accounts = accounts;
    // Emit custom event or call callback
  }

  handleChainChanged(chainId) {
    this.chainId = chainId;
    // Handle chain change
  }

  handleDisconnect() {
    this.accounts = [];
    this.chainId = null;
    // Handle disconnect
  }
}
```

### Framework Integrations

<CodeGroup>

```javascript Next.js Integration
// pages/_app.js
import "@sei-js/sei-global-wallet/eip6963";
import { WalletProvider } from '../components/WalletProvider';

export default function App({ Component, pageProps }) {
  return (
    <WalletProvider>
      <Component {...pageProps} />
    </WalletProvider>
  );
}

// components/WalletProvider.js
import { createContext, useContext, useEffect, useState } from 'react';

const WalletContext = createContext();

export function WalletProvider({ children }) {
  const [isClient, setIsClient] = useState(false);
  const [wallets, setWallets] = useState([]);

  useEffect(() => {
    setIsClient(true);
  }, []);

  useEffect(() => {
    if (!isClient) return;

    const handleAnnouncement = (event) => {
      setWallets(prev => [...prev, event.detail]);
    };

    window.addEventListener('eip6963:announceProvider', handleAnnouncement);
    window.dispatchEvent(new Event('eip6963:requestProvider'));

    return () => {
      window.removeEventListener('eip6963:announceProvider', handleAnnouncement);
    };
  }, [isClient]);

  if (!isClient) return null;

  return (
    <WalletContext.Provider value={{ wallets }}>
      {children}
    </WalletContext.Provider>
  );
}

export const useWallets = () => useContext(WalletContext);
```

## Best Practices

<AccordionGroup>
  <Accordion title="Wallet Detection">
    - Always import the EIP-6963 module before wallet detection
    - Implement proper loading states while waiting for wallet discovery
    - Handle cases where no compatible wallets are found
    - Provide fallback options for users without compatible wallets
  </Accordion>

  <Accordion title="Connection Management">
    - Implement proper error handling for connection failures
    - Store connection state appropriately (avoid sensitive data in localStorage)
    - Handle wallet disconnections gracefully
    - Provide clear feedback to users about connection status
  </Accordion>

  <Accordion title="Transaction Handling">
    - Always validate transaction parameters before sending
    - Implement proper error handling for failed transactions
    - Provide transaction status updates to users
    - Handle network switching when necessary
  </Accordion>

  <Accordion title="User Experience">
    - Provide clear wallet selection interfaces
    - Show wallet branding and icons when available
    - Implement proper loading states during operations
    - Offer help documentation for wallet setup
  </Accordion>

## Troubleshooting

<AccordionGroup>
  <Accordion title="Wallet Not Detected">
    **Problem**: Sei Global Wallet not appearing in wallet list
    
    **Solutions**:
    - Ensure the import statement is at the top of your entry file
    - Check that the wallet supports EIP-6963 standard
    - Verify the wallet is properly installed and enabled
    - Try refreshing the page or restarting the application
  </Accordion>

  <Accordion title="Connection Issues">
    **Problem**: Unable to connect to wallet
    
    **Solutions**:
    - Check that the wallet is unlocked
    - Verify network connectivity
    - Ensure the correct network is selected
    - Try disconnecting and reconnecting
  </Accordion>

  <Accordion title="Transaction Failures">
    **Problem**: Transactions fail to execute
    
    **Solutions**:
    - Verify sufficient balance for transaction and fees
    - Check gas limit and gas price settings
    - Ensure correct network is selected
    - Validate transaction parameters
  </Accordion>
</AccordionGroup>
